fun <T> cached(scope: CoroutineScope = GlobalScope, ttl: Long = 1.H, retryOnFail: Int = 0, delayOnLoadError: Long = 0, errorHandler: (attempt: Int, t: Throwable) -> Unit = { _, _ -> }, loader: () -> T): Delegate<T>

Creates a cached delegate. The value will be refreshed in a regular interval. The retryOnFail, delayOnLoadError and the errorHandler provides some error tolerance.

import org.junit.jupiter.api.Assertions.assertEquals

import java.net.UnknownHostException
import kotlin.test.assertTrue

   val errors = mutableListOf<Throwable>()
var counter = 0
val data by cached(retryOnFail = 10, errorHandler = { _, throwable -> errors.add(throwable) }) {

		1
	else throw UnknownHostException("blah.example.com not known")
}

assertEquals(1, data)
assertEquals(1, data)
assertEquals(3, errors.size)
